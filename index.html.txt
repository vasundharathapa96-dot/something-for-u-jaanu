    // Snowfall animation for final slide
    const canvas = document.getElementById('snowCanvas');
    const ctx = canvas.getContext('2d');
    let w,h,snowflakes,animationActive=false;
    function initSnow(){w=canvas.width=innerWidth;h=canvas.height=innerHeight;snowflakes=[];for(let i=0;i<200;i++){snowflakes.push({x:Math.random()*w,y:Math.random()*h,r:Math.random()*3+1,d:Math.random()+1});}}
    function drawSnow(){ctx.clearRect(0,0,w,h);ctx.fillStyle='white';ctx.beginPath();for(let i=0;i<snowflakes.length;i++){let f=snowflakes[i];ctx.moveTo(f.x,f.y);ctx.arc(f.x,f.y,f.r,0,Math.PI*2,true);}ctx.fill();moveSnow();}
    let angle=0;function moveSnow(){angle+=0.01;for(let i=0;i<snowflakes.length;i++){let f=snowflakes[i];f.y+=Math.pow(f.d,2)+1;f.x+=Math.sin(angle)*0.5;if(f.y>h){snowflakes[i]={x:Math.random()*w,y:0,r:f.r,d:f.d};}}}
    function updateSnow(){if(!animationActive)return;drawSnow();requestAnimationFrame(updateSnow);} 
    function startSnow(){if(animationActive)return;animationActive=true;initSnow();updateSnow();}
    window.addEventListener('resize',()=>{if(animationActive)initSnow();});
